Javascript Guidelines: Strict mode examples 

function foo() {
    "use strict"
    var bar = Math.floor(10.9);
    return bar;
}
var value = foo();
console.log(value);

OR better use IIFE


// IIFE 
(function() {
    "use strict"
    
    var floor = function(value) {
        return Math.floor(value);
    };

    console.log(floor(10.9));

})();
Javascript Guidelines : IIFE 

  (function(){ /* code */ }()); // Crockford recommends this one
  (function(){ /* code */ })(); // But this one works just as well

I prefer the later one
Javascript Guidelines : IIFE 

// IIFE with jQuery
(function(window, document, $, undefined) {
    "use strict"
    
    var floor = function(value) {
        return Math.floor(value);
    };
    console.log(floor(10.9));

})(this, this.document, jQuery);

SO THIS IS HOW I PREFER MY JAVASCRIPT

                        
(function(self) {
  self.fooBar = {
    init: function() {
      // This is where you initialize all your functions
      console.log("this is the first module");
      return true;

    },
    events: function() {
      // This is where you should be mapping your functions to events
      console.log("these are the events we call");
      return true;
    }
  };
})(this._namespace = this._namespace || {});
  • Every function should have a init and events function.
  • All modules need to be namespaced into a global object to avoid naming conflicts.
